#include <windows.h>

// A functional function that safely starts a process using modern APIs
void LaunchTargetProcess()
{
    // Define the application to run (using a standard, safe utility for testing)
    wchar_t commandLine[] = L"notepad.exe";

    STARTUPINFOW si;
    PROCESS_INFORMATION pi;

    // Initialize the structures safely
    ZeroMemory(&si, sizeof(si));
    si.cb = sizeof(si);
    ZeroMemory(&pi, sizeof(pi));

    // Start the child process safely within the current user's context
    if (CreateProcessW(
            NULL,           // Application name
            commandLine,    // Command line arguments
            NULL,           // Process handle not inheritable
            NULL,           // Thread handle not inheritable
            FALSE,          // Set handle inheritance to FALSE
            0,              // No creation flags
            NULL,           // Use parent's environment block
            NULL,           // Use parent's starting directory 
            &si,            // Pointer to STARTUPINFO structure
            &pi             // Pointer to PROCESS_INFORMATION structure
        ))
    {
        // Close handles to avoid resource leaks
        CloseHandle(pi.hProcess);
        CloseHandle(pi.hThread);
    }
}

// Standard entry-point for a Windows DLL
BOOL WINAPI DllMain(HINSTANCE hinstDLL, DWORD fdwReason, LPVOID lpvReserved)
{
    switch (fdwReason)
    {
        case DLL_PROCESS_ATTACH:
            // Disable thread library calls if not needed (optimizes performance)
            DisableThreadLibraryCalls(hinstDLL);
            
            // Execute the function safely
            LaunchTargetProcess();
            break;

        case DLL_PROCESS_DETACH:
            // Cleanup code if necessary
            break;
    }
    
    // Return TRUE to indicate successful initialization
    return TRUE;
}
