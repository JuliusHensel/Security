#!/usr/bin/python3
import lxml.html
import requests
import json

def main():
    # 1. CHANGE: Replace with the actual URL where your HTML table is hosted
    url = 'http://localhost:21053/contact.html#' 
    
    try:
        page = requests.get(url)
        page.raise_for_status() # Ensures the request was successful
    except requests.exceptions.RequestException as e:
        print(f"Error fetching the page: {e}")
        return

    tree = lxml.html.fromstring(page.content)
    
    # 2. CHANGE: Target the table rows (skipping the header row)
    # This selects every <tr> that comes after the first <tr> (headers)
    rows = tree.xpath('//table/tr[position() > 1]')
    
    output = []
    
    # Loop through each row to extract specific column data
    for row in rows:
        # td[1] = Corp Name, td[2] = Corp ID, td[3] = Content, td[4] = Last Update
        corp_name = row.xpath('./td[1]/text()')
        corp_id = row.xpath('./td[2]/text()')
        content = row.xpath('./td[3]/text()')
        last_update = row.xpath('./td[4]/text()')
        
        # Clean up the text (strip whitespaces/newlines) if data exists
        resp = {
            'corp': corp_name[0].strip() if corp_name else None,
            #'corp_id': corp_id[0].strip() if corp_id else None,
            #'content': content[0].strip() if content else None,
            #'last_update': last_update[0].strip() if last_update else None
        }
        output.append(resp)
        
    # Print the beautifully structured JSON output
    print(json.dumps(output, indent=4))

if __name__ == "__main__":
    main()

#!/usr/bin/python3
import requests
import lxml.html
from urllib.parse import urljoin, urlparse

def crawl(start_url):
    visited_pages = set()
    pages_to_visit = [start_url]
    
    target_domain = urlparse(start_url).netloc
    print(f"[*] Starting crawler on domain: {target_domain}\n")

    while pages_to_visit:
        current_url = pages_to_visit.pop(0)

        if current_url in visited_pages:
            continue

        print(f"[+] Crawling: {current_url}")
        
        try:
            response = requests.get(current_url, timeout=5)
            visited_pages.add(current_url)

            if "text/html" not in response.headers.get("Content-Type", ""):
                continue

            # Parse using lxml instead of BeautifulSoup
            tree = lxml.html.fromstring(response.content)
            
            # --- CHECK FOR FLAGS HERE ---
            if "WoFssgA1KWSDRtoWziO4" in response.text:
                 print("[!!!] Found target text on this page!")
            # ----------------------------

            # Find all links using XPath
            links = tree.xpath('//a/@href')
            for href in links:
                absolute_url = urljoin(current_url, href)
                absolute_url = urlparse(absolute_url)._replace(fragment="").geturl()

                if urlparse(absolute_url).netloc == target_domain:
                    if absolute_url not in visited_pages and absolute_url not in pages_to_visit:
                        pages_to_visit.append(absolute_url)

        except requests.exceptions.RequestException as e:
            print(f"[-] Error crawling {current_url}: {e}")

    print("\n[*] Crawl complete!")
    print(f"[*] Total unique pages discovered: {len(visited_pages)}")

if __name__ == "__main__":
    START_URL = "http://localhost:1211"
    crawl(START_URL)

SECRET_20_PATTERN = re.compile(r'(?:[\s\'"]|><)([a-zA-Z0-9]{20})(?:[\s\'"]|><)')

def check_content(html_content, url):
    """Scans the HTML content for the specified target strings."""
    found_something = False

    # 1. Check for password variants
    pass_matches = PASS_PATTERN.findall(html_content)
    if pass_matches:
        print(f"  [!] Found password keyword(s): {set(pass_matches)}")
        found_something = True
        
    # 2. Check for username variants
    user_matches = USER_PATTERN.findall(html_content)
    if user_matches:
        print(f"  [!] Found username keyword(s): {set(user_matches)}")
        found_something = True

        print("  [-] No matches found.")
